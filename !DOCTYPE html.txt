    <script>
        // UI Logics
        function switchTab(tabName) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));
            document.querySelectorAll('.tab-btn').forEach(btn => {
                btn.classList.remove('bg-emerald-600', 'text-white', 'shadow-md', 'shadow-emerald-200');
                btn.classList.add('bg-white', 'text-slate-600', 'border', 'border-slate-200');
            });
            document.getElementById('content-' + tabName).classList.remove('hidden');
            const activeBtn = document.getElementById('tab-' + tabName);
            activeBtn.classList.add('bg-emerald-600', 'text-white', 'shadow-md', 'shadow-emerald-200');
            activeBtn.classList.remove('bg-white', 'text-slate-600', 'border', 'border-slate-200');
        }

        function toggleDonationModal() {
            const modal = document.getElementById('donasi-modal');
            const content = document.getElementById('donasi-modal-content');
            if (modal.classList.contains('hidden')) {
                modal.classList.remove('hidden');
                setTimeout(() => {
                    modal.classList.remove('opacity-0');
                    content.classList.remove('translate-y-full');
                }, 10);
            } else {
                modal.classList.add('opacity-0');
                content.classList.add('translate-y-full');
                setTimeout(() => modal.classList.add('hidden'), 300);
            }
        }

        function showToast(message, type = 'success') {
            const container = document.getElementById('toast-container');
            const toast = document.createElement('div');
            const bgColor = type === 'success' ? 'bg-emerald-600' : 'bg-red-500';
            const icon = type === 'success' ? 'fa-check-circle' : 'fa-circle-exclamation';

            toast.className = `${bgColor} text-white px-4 py-3 rounded-xl shadow-xl flex items-center gap-3 transform transition-all duration-300 -translate-y-10 opacity-0 text-sm font-medium`;
            toast.innerHTML = `<i class="fa-solid ${icon}"></i> <span>${message}</span>`;
            
            container.appendChild(toast);
            setTimeout(() => toast.classList.remove('-translate-y-10', 'opacity-0'), 10);
            setTimeout(() => {
                toast.classList.add('-translate-y-10', 'opacity-0');
                setTimeout(() => toast.remove(), 300);
            }, 3000);
        }

        function formatRupiah(amount) {
            return new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', minimumFractionDigits: 0 }).format(amount);
        }

        function formatDate(dateString) {
            if(!dateString) return "-";
            return new Date(dateString).toLocaleDateString('id-ID', { day: 'numeric', month: 'short', year: 'numeric' });
        }
    </script>

        async function initAuth() {
            try {
                // Menggunakan SignInAnonymously langsung karena Anda menggunakan Firebase Config milik sendiri.
                // Token bawaan (__initial_auth_token) hanya berlaku untuk config default.
                await signInAnonymously(auth);
                setupListeners();
                checkAndSeedData();
            } catch (error) {
                console.error("Auth failed:", error);
            }
        }

        async function checkAndSeedData() {
            try {
                const docSnap = await getDoc(campaignRef);
                if (!docSnap.exists()) {
                    await setDoc(campaignRef, { target: 150000000, collected: 0, lastUpdate: new Date().toISOString() });
                    // Seed initial data
                    await addDoc(donorsColRef, { name: "Hamba Allah", amount: 5000000, date: new Date().toISOString(), message: "Bismillah", status: "approved" });
                    await updateCampaignTotal(5000000);
                }
            } catch (e) {}
        }

        // Helper to update total safely
        async function updateCampaignTotal(amountToAdd) {
            try {
                const snap = await getDoc(campaignRef);
                if (snap.exists()) {
                    const current = snap.data().collected || 0;
                    await updateDoc(campaignRef, {
                        collected: current + amountToAdd,
                        lastUpdate: new Date().toISOString()
                    });
                }
            } catch (error) { console.error("Update total error", error); }
        }

        function setupListeners() {
            // 1. Campaign Data Listener
            onSnapshot(campaignRef, (docSnap) => {
                if (docSnap.exists()) {
                    const data = docSnap.data();
                    const remaining = data.target - data.collected;
                    const percentage = Math.min((data.collected / data.target) * 100, 100);

                    document.getElementById('target-amount').innerText = formatRupiah(data.target);
                    document.getElementById('collected-amount').innerText = formatRupiah(data.collected);
                    document.getElementById('remaining-amount').innerText = formatRupiah(Math.max(0, remaining));
                    document.getElementById('vendor-remaining').innerText = formatRupiah(Math.max(0, remaining));
                    document.getElementById('percent-text').innerText = Math.round(percentage) + '%';
                    document.getElementById('progress-bar').style.width = percentage + '%';
                    
                    if(data.lastUpdate) {
                        document.getElementById('last-update-text').innerText = 'Update: ' + formatDate(data.lastUpdate);
                    }
                }
            }, (e) => {
                console.error("Snapshot Error:", e);
                // Menambahkan notifikasi jika rules firebase belum diatur
                if(e.code === 'permission-denied') {
                    showToast("Error Firebase: Cek Rules (Aturan Keamanan) Anda!", "error");
                    document.getElementById('last-update-text').innerText = "Akses Data Ditolak (Cek Rules)";
                }
            });

            // 2. Donors Listener
            onSnapshot(donorsColRef, (snapshot) => {
                const allDonors = [];
                snapshot.forEach(doc => allDonors.push({ id: doc.id, ...doc.data() }));
                
                // Sort by date desc
                allDonors.sort((a, b) => new Date(b.date) - new Date(a.date));

                const approvedDonors = allDonors.filter(d => d.status === 'approved');

                // Render Public Donors
                const publicContainer = document.getElementById('donors-list-container');
                publicContainer.innerHTML = '';
                if(approvedDonors.length === 0) {
                    publicContainer.innerHTML = `<div class="text-center py-8 text-slate-500 text-sm">Belum ada donasi terkonfirmasi.</div>`;
                } else {
                    approvedDonors.forEach(donor => {
                        const messageHtml = donor.message ? `<p class="text-[11px] text-slate-500 italic mt-1 border-l-2 border-emerald-200 pl-2">"${donor.message}"</p>` : '';
                        publicContainer.innerHTML += `
                            <div class="bg-white p-4 rounded-xl border border-slate-100 shadow-sm flex justify-between items-center relative overflow-hidden">
                                <div class="absolute left-0 top-0 bottom-0 w-1 bg-emerald-500"></div>
                                <div class="pl-2">
                                    <p class="text-sm font-bold text-slate-800">${donor.name}</p>
                                    <p class="text-[10px] text-slate-400 mb-1">${formatDate(donor.date)}</p>
                                    ${messageHtml}
                                </div>
                                <div class="text-right">
                                    <p class="text-sm font-bold text-emerald-600">${formatRupiah(donor.amount)}</p>
                                </div>
                            </div>
                        `;
                    });
                }
            }, (e) => console.error(e));

            // 3. Updates Listener
            onSnapshot(updatesColRef, (snapshot) => {
                const updates = [];
                snapshot.forEach(doc => updates.push({ id: doc.id, ...doc.data() }));
                updates.sort((a, b) => new Date(b.date) - new Date(a.date));

                const timeline = document.getElementById('updates-timeline');
                timeline.innerHTML = '';
                if(updates.length === 0) timeline.innerHTML = `<div class="text-center py-8 text-slate-500 text-sm ml-[-12px]">Belum ada kabar terbaru.</div>`;

                updates.forEach(update => {
                    const imgHtml = update.imageUrl ? `<img src="${update.imageUrl}" class="w-full h-40 object-cover rounded-lg mb-2 border border-slate-200">` : '';
                    timeline.innerHTML += `
                        <div class="relative ml-6">
                            <div class="absolute w-3 h-3 bg-emerald-500 rounded-full -left-[31px] top-1 border-2 border-white"></div>
                            <div class="bg-white p-4 rounded-xl border border-slate-100 shadow-sm relative">

                                ${imgHtml}
                                <p class="text-xs text-slate-600 leading-relaxed">${update.description.replace(/\n/g, '<br>')}</p>
                            </div>
                        </div>
                    `;
                });
            }, (e) => console.error(e));
        }

            e.preventDefault();
            const btn = document.getElementById('btn-submit-donasi');
            btn.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Memproses...';
            btn.disabled = true;

            const name = document.getElementById('user-nama').value;
            const wa = document.getElementById('user-wa').value;
            const amount = parseFloat(document.getElementById('user-nominal').value);
            const message = document.getElementById('user-doa').value;
            // File input logic: Since direct upload to Firebase requires Storage setup, 
            // we rely on WhatsApp for the actual file transfer. The form ensures they selected a file.

            try {
                // 1. Save as Pending
                await addDoc(donorsColRef, {
                    name: name,
                    wa: wa,
                    amount: amount,
                    message: message,
                    date: new Date().toISOString(),
                    status: 'pending' // Requires admin approval
                });

                // 2. Format WA Message
                let waText = `Assalamu'alaikum, saya sudah mengirim infaq untuk pelunasan dinding kaca masjid.%0A%0A*Detail Donatur:*%0ANama: ${name}%0ANominal: ${formatRupiah(amount)}`;
                if(message) waText += `%0ADoa: ${message}`;
                waText += `%0A%0A_(Saya akan melampirkan foto bukti transfer setelah pesan ini)_`;

                // 3. Open WhatsApp & Close Modal
                const waUrl = `https://wa.me/${ADMIN_WA}?text=${waText}`;
                window.open(waUrl, '_blank');
                
                toggleDonationModal();
                document.getElementById('form-user-donasi').reset();
                showToast("Form berhasil dikirim. Silakan lanjut di WhatsApp.");

            } catch (error) {
                console.error("Donation submit error:", error);
                showToast("Terjadi kesalahan sistem", "error");
            } finally {
                btn.innerHTML = 'Lanjut ke WhatsApp <i class="fa-brands fa-whatsapp text-xl"></i>';
                btn.disabled = false;
            }
        });

        // Start 
        if (app) initAuth();

    </script>
